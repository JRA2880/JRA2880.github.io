					* Program Exercise: Write an application named DisplayMultiplicationTable that displays a table of the products of every combination of two integers from 1 through 10.
					* 
					* Pseudocode:
					* 1.) Display to output screen the header row of the following " \t1\t2\t3\t4\t5\t6\t7\t8\t9\t10"
					*                                                                --------------------------------
					* 2.) intialize the following variables:
					* a.) int i;
					* b.  int j;
					* 3.) Establish the outer loop as follows: (for i = 0; i <= 10; ++i)
					* 4.) print to the screen in the outer loop i.
					* 5.) Establish the inner loop as follows: (for j = 0; j <= 10; ++j)
					* 6.) print to the screen in the inner loop the following: "{0}", i * j.
					* 7.) print a WriteLine to the screen.
					* 8.) print a ReadLine to the screen.
					* 9.) Finish the program.
					*/
					using System;
					using System.Collections.Generic;
					using System.Linq;
					using System.Text;
					using System.Threading.Tasks;
					using static System.Console;
					using static System.String;

					namespace DisplayMultiplicationTable
					{
						class DisplayMultiplicationTable
						{
							static void Main()
							{
								int i = 0;
								int j = 0;

								WriteLine("\t1\t2\t3\t4\t5\t6\t7\t8\t9\t10");
								WriteLine("-----------------------------------------------------------------------------------");

								for( i = 1; i <= 10; i++)
								{
									Write(i);

									for(j = 1; j <= 10; j++)
									{
										Write("\t{0}", i * j);
									}
								 WriteLine();
								}
								ReadLine();
							}
						}
					}
